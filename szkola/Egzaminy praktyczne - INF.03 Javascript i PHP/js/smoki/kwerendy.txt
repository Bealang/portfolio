SELECT nazwa, dlugosc, szerokosc FROM smok WHERE pochodzenie = 'Polska';

SELECT DISTINCT pochodzenie FROM smok ORDER BY pochodzenie ASC;

SELECT rok, AVG(dlugosc) AS "Średnia długość"  FROM parada JOIN udzial u ON id = id_parada  JOIN smok ON id = id_smok  WHERE rok > 2005 GROUP BY rok;
